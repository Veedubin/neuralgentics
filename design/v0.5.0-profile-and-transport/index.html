    Section 1: CI -short Fix (T-CI-FIX)¶

    Current State Analysis¶

    Finding: ci.yml line 73 already includes the -short flag:

    // File: .github/workflows/ci.yml, line 68-75
- name: Test all Go modules
  run: |
    MODULES="packages/memory packages/orchestrator-go packages/broker-go packages/backend-go"
    for m in $MODULES; do
      echo "::group::Testing $m"
      (cd "$m" && GOWORK=off go test -short -timeout 60s ./...)
      echo "::endgroup::"
    done

    The -short flag was apparently added in a previous commit (v0.4.0 or earlier). The release.yml has no test step at all — it only builds binaries and pushes containers.

    Root Cause (if CI still times out)¶

    The -short flag works by having tests check testing.Short() and skip themselves. The pre-existing TestBroker integration test in packages/broker-go/src/neuralgentics/broker/broker_integration_test.go (~13 test files in the broker module) spawns subprocesses and uses proxy.readLoop which blocks on bufio.Scanner. If that test does not call t.Skip() when testing.Short() is true, then -short has no effect and the test will still run in CI.

    Fix (Two Options)¶

    Option A (Preferred — No Test Changes): - Verify that broker_integration_test.go checks testing.Short(). If it does, no further changes needed — CI is already fixed. - If it does not, add if testing.Short() { t.Skip("skipping integration test in short mode") } to TestBroker.

    Option B (If -short doesn't apply): - Split the broker test step in CI into two steps: - Unit tests: go test -short -timeout 60s ./... - Integration tests: go test -run TestBroker -timeout 120s ./... (separate step, longer timeout)

    File Changes¶

    File Change Lines
    .github/workflows/ci.yml VERIFY only — already has -short on line 73 None needed
    .github/workflows/release.yml NO test step exists — builds only None needed
    packages/broker-go/*/broker_integration_test.go Add testing.Short() guard if missing ~3 lines

    Decision¶

    Investigation needed during implementation: Confirm that broker_integration_test.go skips long-running tests in -short mode. The CI yml already has the flag; the tests may just need a guard. If already guarded, this card is a no-op and can be marked done.

    Trade-off: opencode.json vs provider-pref.json¶

    • opencode.json (small_model key at line 163) is the source of truth for OpenCode's model dispatch. The TUI does NOT modify this file — OpenCode reads it at startup.
    • provider-pref.json is the TUI's runtime source of truth for provider selection and small model mapping.
    • Rationale: opencode.json is a static config file checked into the repo. provider-pref.json is a per-machine runtime preference. The TUI should not mutate checked-in config.
    • Future: When Neuralgentics adds an /opencode write command that regenerates opencode.json from TUI state, the small_model field will be populated from provider-pref.json.smallModel. Deferred to v0.5.1.

    package proxy

import (
    "bufio"
    "bytes"
    "context"
    "encoding/json"
    "fmt"
    "net/http"
    "strings"
    "time"

    "github.com/Veedubin/neuralgentics-broker/src/neuralgentics/broker/types"

// HTTPClient handles JSON-RPC communication with remote MCP servers over HTTP/SSE.
type HTTPClient struct {
    baseURL    string
    httpClient *http.Client
    sessionID  string
    headers    map[string]string // Authorization, custom headers from Env
}

// NewHTTPClient creates an HTTP client for a remote MCP server.
func NewHTTPClient(baseURL string, headers map[string]string) *HTTPClient {
    return &HTTPClient{
        baseURL: baseURL,
        httpClient: &http.Client{Timeout: 30 * time.Second},
        headers:   headers,
    }
}

// Initialize performs the MCP initialize handshake over HTTP POST.
func (h *HTTPClient) Initialize(ctx context.Context) error {
    params := map[string]any{
        "protocolVersion": "2024-11-05",
        "capabilities":    map[string]any{},
        "clientInfo": map[string]any{
            "name":    "neuralgentics-broker",
            "version": "0.5.0",
        },
    }
    resp, err := h.sendRPC(ctx, "initialize", params)
    if err != nil {
        return fmt.Errorf("http initialize handshake: %w", err)
    }
    if resp.Error != nil {
        return fmt.Errorf("http initialize error: code=%d message=%s", resp.Error.Code, resp.Error.Message)
    }
    // Send initialized notification (fire-and-forget).
    _, _ = h.sendRPC(ctx, "notifications/initialized", nil)
    return nil
}

// ListTools requests the tool list from the remote server.
func (h *HTTPClient) ListTools(ctx context.Context) ([]types.ToolSummary, error) {
    resp, err := h.sendRPC(ctx, "tools/list", map[string]any{})
    if err != nil {
        return nil, fmt.Errorf("http list tools: %w", err)
    }
    if resp.Error != nil {
        return nil, fmt.Errorf("http list tools error: code=%d message=%s", resp.Error.Code, resp.Error.Message)
    }
    var result struct {
        Tools []struct {
            Name        string `json:"name"`
            Description string `json:"description"`
        } `json:"tools"`
    }
    if err := json.Unmarshal(resp.Result, &result); err != nil {
        return nil, fmt.Errorf("unmarshal tools: %w", err)
    }
    summaries := make([]types.ToolSummary, 0, len(result.Tools))
    for _, t := range result.Tools {
        summaries = append(summaries, types.ToolSummary{
            Server:      h.baseURL,
            Name:        t.Name,
            Description: t.Description,
        })
    }
    return summaries, nil
}

// Call sends a JSON-RPC method call and returns the result.
func (h *HTTPClient) Call(ctx context.Context, method string, params map[string]any) (map[string]any, error) {
    resp, err := h.sendRPC(ctx, method, params)
    if err != nil {
        return nil, fmt.Errorf("http call %q: %w", method, err)
    }
    if resp.Error != nil {
        return nil, fmt.Errorf("http error: code=%d message=%s", resp.Error.Code, resp.Error.Message)
    }
    var result map[string]any
    if err := json.Unmarshal(resp.Result, &result); err != nil {
        return nil, fmt.Errorf("unmarshal result: %w", err)
    }
    return result, nil
}

// CallSSE opens an SSE connection and reads event-stream lines.
// Uses the standard SSE wire format: "data: <json>\n\n".
func (h *HTTPClient) CallSSE(ctx context.Context, method string, params map[string]any) (map[string]any, error) {
    reqBody, err := json.Marshal(jsonrpcRequest{
        JSONRPC: "2.0",
        ID:      1,
        Method:  method,
        Params:  params,
    })
    if err != nil {
        return nil, fmt.Errorf("marshal SSE request: %w", err)
    }

    httpReq, err := http.NewRequestWithContext(ctx, http.MethodPost, h.baseURL, bytes.NewReader(reqBody))
    if err != nil {
        return nil, fmt.Errorf("create SSE request: %w", err)
    }
    httpReq.Header.Set("Content-Type", "application/json")
    httpReq.Header.Set("Accept", "text/event-stream")
    for k, v := range h.headers {
        httpReq.Header.Set(k, v)
    }

    resp, err := h.httpClient.Do(httpReq)
    if err != nil {
        return nil, fmt.Errorf("SSE request: %w", err)
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        return nil, fmt.Errorf("SSE request returned %d", resp.StatusCode)
    }

    // Parse SSE event stream.
    scanner := bufio.NewScanner(resp.Body)
    scanner.Buffer(make([]byte, 0, 64*1024), 1024*1024)
    var lastData []byte
    for scanner.Scan() {
        line := scanner.Text()
        if strings.HasPrefix(line, "data: ") {
            lastData = []byte(strings.TrimPrefix(line, "data: "))
        }
        if line == "" && len(lastData) > 0 {
            // End of event — parse the accumulated data.
            break
        }
    }
    if err := scanner.Err(); err != nil {
        return nil, fmt.Errorf("SSE scan error: %w", err)
    }
    if len(lastData) == 0 {
        return nil, fmt.Errorf("no SSE data received")
    }

    var rpcResp jsonrpcResponse
    if err := json.Unmarshal(lastData, &rpcResp); err != nil {
        return nil, fmt.Errorf("unmarshal SSE response: %w", err)
    }
    if rpcResp.Error != nil {
        return nil, fmt.Errorf("SSE error: code=%d message=%s", rpcResp.Error.Code, rpcResp.Error.Message)
    }
    var result map[string]any
    if err := json.Unmarshal(rpcResp.Result, &result); err != nil {
        return nil, fmt.Errorf("unmarshal SSE result: %w", err)
    }
    return result, nil
}

// sendRPC marshals, sends, and parses a JSON-RPC request/response over HTTP POST.
func (h *HTTPClient) sendRPC(ctx context.Context, method string, params map[string]any) (*jsonrpcResponse, error) {
    req := jsonrpcRequest{
        JSONRPC: "2.0",
        ID:      1,
        Method:  method,
        Params:  params,
    }
    body, err := json.Marshal(req)
    if err != nil {
        return nil, fmt.Errorf("marshal request: %w", err)
    }

    httpReq, err := http.NewRequestWithContext(ctx, http.MethodPost, h.baseURL, bytes.NewReader(body))
    if err != nil {
        return nil, fmt.Errorf("create request: %w", err)
    }
    httpReq.Header.Set("Content-Type", "application/json")
    for k, v := range h.headers {
        httpReq.Header.Set(k, v)
    }

    httpResp, err := h.httpClient.Do(httpReq)
    if err != nil {
        return nil, fmt.Errorf("http request: %w", err)
    }
    defer httpResp.Body.Close()

    if httpResp.StatusCode != http.StatusOK {
        return nil, fmt.Errorf("http %d", httpResp.StatusCode)
    }

    var rpcResp jsonrpcResponse
    if err := json.NewDecoder(httpResp.Body).Decode(&rpcResp); err != nil {
        return nil, fmt.Errorf("decode response: %w", err)
    }
    return &rpcResp, nil
}

type Client interface {

// GetClient returns a Client appropriate for the given ServerConfig's transport type.
func GetClient(config types.ServerConfig) (Client, error) {
    switch config.Type {
    case "http", "sse":
        url := config.Env["NEURALGENTICS_MCP_URL"]
        if url == "" {
            return nil, fmt.Errorf("http transport requires NEURALGENTICS_MCP_URL in env")
        }
        headers := make(map[string]string)
        if auth, ok := config.Env["Authorization"]; ok {
            headers["Authorization"] = auth
        }
        return &httpClientAdapter{
            client: NewHTTPClient(url, headers),
        }, nil
    case "stdio":
        return &stdioClientAdapter{
            proxy: NewMCPProxy(),
        }, nil // Note: the adapter's Initialize will call proxy.Initialize
    default:
        return nil, fmt.Errorf("unknown transport type: %q", config.Type)
    }
}

    Adapter types (lives in proxy.go to avoid circular imports):

    // httpClientAdapter wraps HTTPClient to implement the Client interface.

func (a *httpClientAdapter) Initialize(_ string) error {

func (b *Broker) startHTTPServer(name string, entry *registry.ServerEntry) error {
    url := entry.Config.Env["NEURALGENTICS_MCP_URL"]
    if url == "" {
        return fmt.Errorf("server %q has no NEURALGENTICS_MCP_URL", name)
    }
    headers := make(map[string]string)
    if auth, ok := entry.Config.Env["Authorization"]; ok {
        headers["Authorization"] = auth
    }
    httpClient := proxy.NewHTTPClient(url, headers)

    // Initialize handshake.
    if err := httpClient.Initialize(context.Background()); err != nil {
        return fmt.Errorf("initialize http server %q: %w", name, err)
    }

    // Discover tools.
    tools, err := httpClient.ListTools(context.Background())
    if err != nil {
        return fmt.Errorf("list tools from http %q: %w", name, err)
    }
    // Fix server names in summaries.
    for i := range tools {
        tools[i].Server = name
    }
    entry.Tools = tools
    b.registry.UpdateEntry(name, entry)
    return nil
}

    Section 4: OCI-Shareable Profile Export/Import (T-PROFILE-OCI)¶

    4.1 File Format¶

    Archive: neuralgentics-profile-{timestamp}.tar.gz

    Internal structure: 

    profile.json           # Active provider, small model, version

// Profile contains all serialized components of an export.

// Sign computes HMAC-SHA256 of concatenated profile data.
func Sign(profileBytes []byte, passphrase string) ([]byte, error) {
    mac := hmac.New(sha256.New, []byte(passphrase))
    mac.Write(profileBytes)
    return mac.Sum(nil), nil
}

// Verify checks an HMAC-SHA256 signature against the profile data.
func Verify(profileBytes, signature []byte, passphrase string) bool {
    expected, err := Sign(profileBytes, passphrase)
    if err != nil {
        return false
    }
    return hmac.Equal(expected, signature)
}

func writeTarGz(w io.Writer, profile Profile, passphrase string) error {
    gz := gzip.NewWriter(w)
    defer gz.Close()
    tw := tar.NewWriter(gz)
    defer tw.Close()

    files := map[string][]byte{
        "manifest.json":         profile.Manifest,
        "provider-pref.json":    profile.Pref,
        "catalog.lock.json":     profile.Catalog,
        "permissions.json":      profile.Permissions,
        "opencode.snapshot.json": profile.Opencode,
    }

    // Concatenate all content for signing.
    var checksumBuf bytes.Buffer
    for _, name := range []string{"manifest.json", "provider-pref.json", "catalog.lock.json", "permissions.json", "opencode.snapshot.json"} {
        data := files[name]
        checksumBuf.Write(data)
        hdr := &tar.Header{
            Name: name,
            Size: int64(len(data)),
            Mode: 0644,
        }
        if err := tw.WriteHeader(hdr); err != nil {
            return fmt.Errorf("write header for %s: %w", name, err)
        }
        if _, err := tw.Write(data); err != nil {
            return fmt.Errorf("write data for %s: %w", name, err)
        }
    }

    // Compute checksum.
    sha := sha256.Sum256(checksumBuf.Bytes())
    checksum := "sha256:" + hex.EncodeToString(sha[:])

    // Sign if passphrase provided.
    if passphrase != "" {
        sig, err := Sign(checksumBuf.Bytes(), passphrase)
        if err != nil {
            return fmt.Errorf("sign profile: %w", err)
        }
        sigHdr := &tar.Header{
            Name: "signature.bin",
            Size: int64(len(sig)),
            Mode: 0644,
        }
        if err := tw.WriteHeader(sigHdr); err != nil {
            return fmt.Errorf("write signature header: %w", err)
        }
        if _, err := tw.Write(sig); err != nil {
            return fmt.Errorf("write signature: %w", err)
        }
    }

    return nil
}

func providerPrefPath() string {
    xdg := os.Getenv("XDG_CONFIG_HOME")
    if xdg == "" {
        xdg = os.Getenv("HOME") + "/.config"
    }
    return xdg + "/neuralgentics/provider-pref.json"
}
